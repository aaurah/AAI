The theme change button is typically located in the settings or preferences section of an application.

To add a theme change button to your AAI application, you can follow these steps:

1. Create a new button element in your HTML file.
2. Add an event listener to the button that calls a function to change the theme.
3. Define the function that changes the theme.

Here's an example of how you could implement a theme change button in JavaScript:

javascript

// Get the theme buttons
const themeButtons = document.querySelectorAll('.theme-button');

// Add an event listener to each theme button
themeButtons.forEach(button => {
  button.addEventListener('click', () => {
    // Get the theme name from the button
    const themeName = button.getAttribute('data-theme');

    // Change the theme
    changeTheme(themeName);
  });
});

// Define the change theme function
function changeTheme(themeName) {
  // Remove the active theme class from all buttons
  document.querySelectorAll('.theme-button').forEach(button => {
    button.classList.remove('active');
  });

  // Add the active theme class to the selected button
  document.querySelector(`.theme-button[data-theme="${themeName}"]`).classList.add('active');

  // Change the theme
  switch (themeName) {
    case 'amoled-black':
      document.body.classList.add('amoled-black');
      break;
    case 'dark':
      document.body.classList.add('dark');
      break;
    case 'light':
      document.body.classList.add('light');
      break;
    case 'night-yellow':
      document.body.classList.add('night-yellow');
      break;
  }
}
